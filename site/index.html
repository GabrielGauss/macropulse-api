Integration

One endpoint. Full macro context.

Plug the regime signal into your existing pipeline in under 60 seconds. No infra changes — just an HTTP call.

python 
# Get the current macro regime signal
import requests

resp = requests.get(
    "https://api.macropulse.live/v1/signals/latest",
    headers={"X-MacroPulse-Key": "mp_free_..."}
)
signal = resp.json()

regime  = signal["regime"]["most_likely"]   # "recovery"
conf    = signal["regime"]["confidence"]     # "HIGH"
persist = signal["regime"]["persistence_days"] # 18
zscore  = signal["net_liquidity"]["zscore"]    # -0.40

# Regime-aware model weighting
EXPOSURE = {
    "expansion":  1.00,
    "recovery":   0.75,
    "tightening": 0.25,
    "risk_off":   0.00,
}
weight = EXPOSURE[regime]  # → 0.75
