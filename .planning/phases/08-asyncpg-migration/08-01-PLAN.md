from database.connection import get_db_conn
# get_db_conn() is an asynccontextmanager yielding asyncpg.Connection
# conn.execute(sql, *args) -> None
# conn.fetchrow(sql, *args) -> asyncpg.Record | None
# conn.fetch(sql, *args) -> list[asyncpg.Record]
# conn.transaction() -> async context manager for explicit transactions

<!-- asyncpg key rules -->
# Parameters: $1, $2, $3 (positional only — NO %s or %(name)s)
# conn.execute(sql, arg1, arg2, ...)  -- NOT conn.execute(sql, {"key": val})
# fetchrow returns asyncpg.Record: wrap with dict(row) if row else None
# fetch returns list[Record]: wrap with [dict(r) for r in rows]
# asyncpg.Record supports row["col"] but NOT row.get("col") — always return dict
</interfaces>

FUNCTION CONVERSION PATTERN (apply to every function):

    def get_user_by_email(email: str) -> dict[str, Any] | None:

        with get_sync_cursor() as cur:

  AFTER:
    async def get_user_by_email(email: str) -> dict[str, Any] | None:
        sql = "SELECT ... WHERE email = $1;"
        async with get_db_conn() as conn:
            row = await conn.fetchrow(sql, email)
            return dict(row) if row else None

SPECIFIC PATTERNS BY RETURN TYPE:

1. Single-row writes (execute, no return):
   async with get_db_conn() as conn:
       await conn.execute(sql, arg1, arg2, ...)

2. Single-row reads (fetchone equivalent):
   async with get_db_conn() as conn:
       row = await conn.fetchrow(sql, arg1, ...)
       return dict(row) if row else None

3. Multi-row reads (fetchall equivalent):
   async with get_db_conn() as conn:
       rows = await conn.fetch(sql, arg1, ...)
       return [dict(r) for r in rows]

4. RETURNING clause (single row):
   async with get_db_conn() as conn:
       row = await conn.fetchrow(sql, arg1, ...)
       return dict(row)   # RETURNING always succeeds or raises

5. fetch_subscriber_emails (returns list[str], not list[dict]):
   async with get_db_conn() as conn:
       rows = await conn.fetch(sql)
       return [row["email"] for row in rows]
