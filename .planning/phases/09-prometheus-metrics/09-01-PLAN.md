    ```python
    async def check_pipeline_staleness():
        """Alert via Brevo email if pipeline hasn't run successfully in >26h."""
        import asyncio
        from database.connection import get_db_conn
        from database import queries

        try:
            async with get_db_conn(pool) as conn:
                last_run = await queries.fetch_latest_pipeline_run(conn)

            if last_run is None:
                return  # No runs yet — skip

            age_hours = (datetime.utcnow() - last_run["completed_at"]).total_seconds() / 3600
            if age_hours > 26:
                # Send Brevo alert
                await send_staleness_alert(age_hours)
        except Exception as e:
            logger.error(f"Staleness check failed: {e}")
    ```
